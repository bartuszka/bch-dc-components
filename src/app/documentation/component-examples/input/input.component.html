Input field API

TypeScript

    import { BchDcInputModule } from 'bch-dc-components';

    public inputForm: FormGroup;
    public inputValue: string;

    public ngOnInit(): void {
      this.inputForm = new FormGroup<InputFormGroup>({
        input: new FormControl('input value'),
        inputDisabled: new FormControl('Input disabled value')
      })

      this.inputValue = this.inputForm.value.input;
    }

    public valueChanged(value: string): void {
      this.inputValue = value;
    }

HTML

      <form [formGroup]="inputForm">
        <bch-dc-input
                      idForLabel="inputField"
                      labelMessage="Input"
                      type="text"
                      formControlName="input"
                      (onValueChanged)="valueChanged($event)">
        </bch-dc-input>

        <bch-dc-input
                      idForLabel="inputFieldDisabled"
                      labelMessage="Disabled Input"
                      [inputDisabled]="true"
                      type="text"
                      formControlName="inputDisabled">
        </bch-dc-input>
      </form>
