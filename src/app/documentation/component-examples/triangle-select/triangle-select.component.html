TypeScript

    import { BchDcTriangleSelectModule } from 'bch-dc-components';}

    export interface TriangleValue {
        name: string,
        value: number
    }

    export interface WordTypeForm {
        wordType: FormControl<TriangleValue>;
    }

    public wordTypeForm: FormGroup<WordTypeForm>;
    public availableValues: TriangleValue[] = [
      {
      name: 'VERB',
      value: 1
      },
      {
      name: 'NOUN',
      value: 2
      },
      {
      name: 'OTHER',
      value: 3
      }
    ];
    public buttonMessages: string[] = ['Verb', 'Noun', 'Other'];
    public selectedValue: TriangleValue = this.availableValues[2];

    private formInitialValue: TriangleValue = {
      name: 'NOUN',
      value: 2
    };

    public valueChanged(value: TriangleValue): void {
      this.selectedValue = value;
    }

    public ngOnInit(): void {
      this.wordTypeForm = new FormGroup<WordTypeForm>({
        wordType: new FormControl(this.formInitialValue)
      });
    }

HTML

    <div class="examples-container">
      <form [formGroup]="wordTypeForm">
        <bch-dc-triangle-select [availableValues]="availableValues"
                                    [buttonMessages]="buttonMessages"
                                    formControlName="wordType">
        </bch-dc-triangle-select>
      </form>

      <bch-dc-triangle-select [value]="availableValues[2]"
                                  [availableValues]="availableValues"
                                  [buttonMessages]="buttonMessages"
                                  (onValueChanged)="valueChanged($event)">
      </bch-dc-triangle-select>
    </div>
